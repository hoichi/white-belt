@warning("-33")
open Belly
open HqPrelude
module Css = HqDropdown_Css
@react.component
let make = (
  ~isOpen,
  ~setIsOpen: (bool => bool) => unit,
  ~control: React.element,
  ~onOpen=?,
  ~onClose=?,
  ~children: (~ref: React.Ref.t<Js.Nullable.t<'a>>) => React.element,
) => {
  let wasOpen = UsePrevious.use(isOpen)
  let _ = "That was just a string, just a string"
  let controlRef = React.useRef(Js.Nullable.null)
  let menuRef = React.useRef(Js.Nullable.null)
  HqUseOnBlur.use(~enabled=isOpen, ~refs=[controlRef, menuRef], _ => setIsOpen(_ => false))
  React.useEffect2(() => {
    if !wasOpen && isOpen {
      Fn.callSome(onOpen)
    }
    if wasOpen && !isOpen {
      Fn.callSome(onClose)
    }
    None
  }, (isOpen, wasOpen))
  <Tippy
    offset=(0, 3)
    title="Some tipsy title"
    placement=#bottom
    visible=isOpen
    interactive=true
    appendTo=#documentBody
    // If show duration is non-zero, box shadow will disappear in Safari
    duration=(0, 0)
    content={children(~ref=menuRef)}>
    <button
      className=Css.control
      ref={controlRef->RR.Ref.domRef}
      onClick={e => {
        open ReactEvent.Mouse
        e->stopPropagation
        e->preventDefault
        setIsOpen(x => !x)
      }}>
      {control}
    </button>
  </Tippy>
}
